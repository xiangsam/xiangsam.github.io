表达式计算

关于中缀表达式，后缀表达式以及表达式树。

中缀表达式计算：
step:

  • 建立两个栈分别存储运算符与数字
  • 入栈，运算符入栈时若需出栈直接与数字进行计算，结果入数字栈
  • 清空运算符栈得结果

后缀表达式计算：
step:

  • 建数字栈
  • 遇到运算符出栈计算，结果入栈

中缀表达式转表达式树：
step:

  • 建立两个栈存储运算符与树结点
  • 遇到数字建立结点入树结点栈
  • 遇到运算符入运算符栈，若需出栈则出栈运算符建立结点并与结点栈两结点连接，连接得到结点入结点栈
  • 清空运算符栈

后缀表达式转表达式树：
step:

  • 建立树节点栈
  • 遇到数字建立结点入栈
  • 遇到运算符建立结点并与栈顶两结点连接，连接结果入栈

表达式树转后缀表达式：
step:

  • 后序编历

表达式树转中缀表达式：
step:

  • 中序遍历，注意添加括号
    • 如果根节点是+号，那么无论左子树以及右子树是什么操作符，它们都是不需要加括号的，因为根节点+号是最小优先级的
    • 如果根节点是-号，那么只有右子树是+号或者-号时，右子树才需要加括号
    • 如果根节点是*号，那么只有左子树或右子树是+号或者-号时，它们才需要加括号
    • 如果根节点是/号，那么如果左子树或右子树是+号或者-号时，它们需要加括号，其次，如果右子树是*号或者/号时，右子树也需要加括号
      void dfs(bt* q)
{
	if(q == NULL)
		return;
	else{
		if(q->op == '+'){
			dfs(q->lchild);
			printf("%c",q->op);
			dfs(q->rchild);
		}
		else if(q->op == '-'){
			dfs(q->lchild);
			printf("%c",q->op);
			if(q->rchild!=NULL&&(q->rchild->op=='+'||q->rchild->op=='-')){
				printf("(");
				dfs(q->rchild);
				printf(")");
			}else{
				dfs(q->rchild);
			}
		}else if(q->op == '*'){
			if(q->lchild!=NULL&&(q->lchild->op=='+'||q->lchild->op=='-')){
				printf("(");
				dfs(q->lchild);
				printf(")");
			}else{
				dfs(q->lchild);
			}printf("%c",q->op);
			if(q->rchild!=NULL&&(q->rchild->op=='+'||q->rchild->op=='-')){
				printf("(");
				dfs(q->rchild);
				printf(")");
			}else{
				dfs(q->rchild);
			}
		}else if(q->op=='/'){
			if(q->lchild!=NULL&&(q->lchild->op=='+'||q->lchild->op=='-')){
				printf("(");
				dfs(q->lchild);
				printf(")");
			}else{
				dfs(q->lchild);
			}printf("%c",q->op);
			if(q->rchild!=NULL&&(q->rchild->op == '+'||q->rchild->op=='-'||q->rchild->op=='*'||q->rchild->op == '/')){
				printf("(");
				dfs(q->rchild);
				printf(")");
			}else{
				dfs(q->rchild);
			}
		}else{
			dfs(q->lchild);
			printf("%d",q->num);
			dfs(q->rchild);
		}
	}
}
